async function uploadPhoto(file){

}

async function doCheckIn(){
  if(!state.checkInPhoto)return;
  const note=document.getElementById("checkInNote")?.value||"";
  const me=state.loginUser;
  const t=now.getHours()*60+now.getMinutes();
  const st=t>11*60+30?"มาสาย":"เข้างานแล้ว";
  const s=staffList.find(x=>x.id===me.id);
  if(s){s.checkIn=timeStr;s.status=st;}
  state.syncing=true;render();
  const photoUrl=await uploadPhoto(dataURLtoFile(state.checkInPhoto,"checkin.jpg"));
  await sendToSheets("checkIn",{date:now.toISOString().slice(0,10),name:me.name,role:me.role,checkInTime:timeStr,status:st,photoUrl:photoUrl||"",note});
  state.syncing=false;
  state.notifications.unshift({id:Date.now(),msg:`${me.name} เช็คอิน ${timeStr}`,time:timeStr,read:false,urgent:st==="มาสาย"});
  render();
}

async function doCheckOut(){
  if(!state.checkOutPhoto)return;
  const me=state.loginUser;
  const s=staffList.find(x=>x.id===me.id);
  if(s){s.checkOut=timeStr;s.status="เลิกงานแล้ว";}
  state.syncing=true;render();
  const photoUrl=await uploadPhoto(dataURLtoFile(state.checkOutPhoto,"checkout.jpg"));
  await sendToSheets("checkOut",{date:now.toISOString().slice(0,10),name:me.name,checkOutTime:timeStr,status:"เลิกงานแล้ว",photoUrl:photoUrl||""});
  state.syncing=false;render();
}

async function submitTask(id){
  if(!state.taskPhotoUrl&&!state.taskPhotoPreview)return;
  const note=document.getElementById("taskNote")?.value||"";
  const ai=state.taskAI;
  tasks=tasks.map(t=>t.id===id?{...t,status:"รอตรวจ",submittedAt:timeStr,note,photoUrl:state.taskPhotoUrl,aiScore:ai?.score,aiStatus:ai?.status,aiNote:ai?.note}:t);
  state.syncing=true;
  const task=tasks.find(t=>t.id===id);
  await sendToSheets("submitTask",{date:now.toISOString().slice(0,10),taskName:task.name,assignee:task.assignee,round:task.round,deadline:task.deadline,submittedAt:timeStr,status:"รอตรวจ",photoUrl:state.taskPhotoUrl||"",aiScore:ai?.score,aiStatus:ai?.status,note});

}

function dataURLtoFile(dataurl,filename){
  const arr=dataurl.split(","),mime=arr[0].match(/:(.*?);/)[1];
  const bstr=atob(arr[1]);let n=bstr.length;const u8arr=new Uint8Array(n);
  while(n--){u8arr[n]=bstr.charCodeAt(n);}
  return new File([u8arr],filename,{type:mime});
}
